To use generics implementations, use "#define" to define T as the type used for the implementation and TN as the suffix name for the methods' and structure's name used for type T.

Example: Use linked list for unsigned long long type

-------------------------------------------------------

#define T unsigned long long
#define TN ull
#include "linkedlist.h"
#undef T
#undef TN

int main() {
    unsigned long long answer_to_everything = 42;
    lnode_ull* sample_node = new_lnode_ull(42);
    free_lnode_ull(sample_node);
    return 0;
}
